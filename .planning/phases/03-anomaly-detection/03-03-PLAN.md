   ```typescript
   import { pool } from './index';
   import type { WatchlistEntry } from '@/types/anomaly';

   export async function addToWatchlist(userId: string, imo: string, notes?: string): Promise<void> {
     await pool.query(
       `INSERT INTO watchlist (user_id, imo, notes)
        VALUES ($1, $2, $3)
        ON CONFLICT (user_id, imo) DO UPDATE SET notes = EXCLUDED.notes`,
       [userId, imo, notes || null]
     );
   }

   export async function removeFromWatchlist(userId: string, imo: string): Promise<void> {
     await pool.query(
       `DELETE FROM watchlist WHERE user_id = $1 AND imo = $2`,
       [userId, imo]
     );
   }

   export async function getUserWatchlist(userId: string): Promise<WatchlistEntry[]> {
     const result = await pool.query<WatchlistEntry>(
       `SELECT user_id as "userId", imo, added_at as "addedAt", notes
        FROM watchlist WHERE user_id = $1 ORDER BY added_at DESC`,
       [userId]
     );
     return result.rows;
   }

   export async function isOnWatchlist(userId: string, imo: string): Promise<boolean> {
     const result = await pool.query(
       `SELECT 1 FROM watchlist WHERE user_id = $1 AND imo = $2`,
       [userId, imo]
     );
     return result.rows.length > 0;
   }

   // Extended query with vessel info for API response
   export async function getWatchlistWithVessels(userId: string) {
     const result = await pool.query(
       `SELECT w.user_id as "userId", w.imo, w.added_at as "addedAt", w.notes,
               v.name as "vesselName", v.flag, v.ship_type as "shipType"
        FROM watchlist w
        LEFT JOIN vessels v ON v.imo = w.imo
        WHERE w.user_id = $1
        ORDER BY w.added_at DESC`,
       [userId]
     );
     return result.rows;
   }
   ```

   ```typescript
   import { pool } from './index';
   import type { Alert } from '@/types/anomaly';

   export async function createAlert(
     userId: string,
     imo: string,
     alertType: string,
     details: object
   ): Promise<void> {
     await pool.query(
       `INSERT INTO alerts (user_id, imo, alert_type, details)
        VALUES ($1, $2, $3, $4)`,
       [userId, imo, alertType, JSON.stringify(details)]
     );
   }

   export async function getUnreadAlerts(userId: string): Promise<Alert[]> {
     const result = await pool.query<Alert>(
       `SELECT id, user_id as "userId", imo, alert_type as "alertType",
               triggered_at as "triggeredAt", read_at as "readAt", details
        FROM alerts
        WHERE user_id = $1 AND read_at IS NULL
        ORDER BY triggered_at DESC`,
       [userId]
     );
     return result.rows;
   }

   export async function getAllAlerts(userId: string, limit = 50): Promise<Alert[]> {
     const result = await pool.query<Alert>(
       `SELECT id, user_id as "userId", imo, alert_type as "alertType",
               triggered_at as "triggeredAt", read_at as "readAt", details
        FROM alerts
        WHERE user_id = $1
        ORDER BY triggered_at DESC
        LIMIT $2`,
       [userId, limit]
     );
     return result.rows;
   }

   export async function markAlertAsRead(alertId: number): Promise<void> {
     await pool.query(
       `UPDATE alerts SET read_at = NOW() WHERE id = $1`,
       [alertId]
     );
   }

   export async function generateAlertsForAnomaly(
     imo: string,
     anomalyType: string,
     details: object
   ): Promise<void> {
     // Create alerts for all users who are watching this vessel
     await pool.query(
       `INSERT INTO alerts (user_id, imo, alert_type, details)
        SELECT user_id, $1, $2, $3
        FROM watchlist WHERE imo = $1`,
       [imo, anomalyType, JSON.stringify(details)]
     );
   }

   // Extended query with vessel names for API
   export async function getAlertsWithVessels(userId: string, limit = 50) {
     const result = await pool.query(
       `SELECT a.id, a.imo, a.alert_type as "alertType",
               a.triggered_at as "triggeredAt", a.read_at as "readAt", a.details,
               v.name as "vesselName", v.flag
        FROM alerts a
        LEFT JOIN vessels v ON v.imo = a.imo
        WHERE a.user_id = $1
        ORDER BY a.triggered_at DESC
        LIMIT $2`,
       [userId, limit]
     );
     return result.rows;
   }
   ```
