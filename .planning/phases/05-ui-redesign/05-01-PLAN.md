Replace the current globals.css (which only has the two @import lines) with the terminal design token block.

Final file content:

```css
@import 'mapbox-gl/dist/mapbox-gl.css';
@import "tailwindcss";

@theme {
  /* Zero out all border-radius utilities to enforce terminal aesthetic.
     After this, rounded-* classes produce no output anywhere in the app. */
  --radius-*: initial;

  /* Register JetBrains Mono as the mono font stack.
     layout.tsx loads the font as --font-jetbrains; this @theme entry
     maps the Tailwind font-mono utility to it. */
  --font-mono: var(--font-jetbrains), ui-monospace, SFMono-Regular, monospace;
}
```

The import order matters: mapbox-gl CSS first, then tailwindcss, then @theme. Do NOT reorder.

Do NOT add --color-amber-500 or --color-black overrides — Tailwind v4 defaults for amber-500 (#f59e0b) and black (#000000) are already correct. Use bg-black, text-amber-500, border-amber-500 directly in components without any color override here.

```tsx
import type { Metadata } from 'next';
import { JetBrains_Mono } from 'next/font/google';

const jetbrainsMono = JetBrains_Mono({
  subsets: ['latin'],
  variable: '--font-jetbrains',
  display: 'swap',
});

  title: 'Tanker Tracker',
  description: 'Real-time Middle East oil tanker tracking',
};

export default function RootLayout({
  children,
}: Readonly<{
  children: React.ReactNode;
}>) {
  return (
    <html lang="en" className={`dark ${jetbrainsMono.variable}`}>
      <body className="bg-black text-white antialiased">
        {children}
      </body>
    </html>
  );
}
```
